namespace Cachopo.CodeTemplates.Tests.Integration.Code
{
    using System;

    /// <summary>
    /// This is an example of a class properly formatted.
    /// </summary>
    public class Example
    {
        private int _exampleProperty;
        private DateTime _someTime;

        public void FirstFunction(DateTime someTime)
        {
            _someTime = someTime;
            _exampleProperty = 25;
        }

        public float SecondFunction()
        {
            var result = _someTime.Millisecond * _exampleProperty;
            return result / 2;
        }
    }
}
